import Alert from "../components/P2Alert";
import { Meta, Story, Canvas, ArgsTable } from "@storybook/blocks";

<Meta title="Components/Alert" component={Alert} />

<h1>Alert</h1>
<p>
  Alert bars are designed to catch the visitor's attention and provide quick and
  concise information about a situation that may require their immediate
  attention.
</p>
<h4>Usage</h4>
<ul>
  <li>
    Success - use to let users know that they have successfully completed an
    action.
  </li>
  <li>
    Warning - use this to inform user of a significant change that will take
    place before they perform an action. This helps users to anticipate that
    they need to pay attention before they proceed to take action.
  </li>
  <li>
    Danger - use to alert users that something has gone wrong or if they have
    tried to do anything but a problem occurred during the session
  </li>
</ul>
